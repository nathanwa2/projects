Step 1 — Create tables
Go to supabase.com → new project → SQL Editor → paste this:
create table check_ins ( id uuid default gen_random_uuid() primary key, date date not null unique, notes text, created_at timestamptz default now() ); create table food_logs ( id uuid default gen_random_uuid() primary key, date date not null, meal_name text not null, calories int default 0, protein_g numeric default 0, carbs_g numeric default 0, fat_g numeric default 0, created_at timestamptz default now() ); create table lift_logs ( id uuid default gen_random_uuid() primary key, date date not null, exercise text not null, weight_kg numeric not null, sets int default 1, reps int default 1, notes text, created_at timestamptz default now() ); -- allow public read/write (RLS off for personal use) alter table check_ins enable row level security; alter table food_logs enable row level security; alter table lift_logs enable row level security; create policy "open" on check_ins for all using (true) with check (true); create policy "open" on food_logs for all using (true) with check (true); create policy "open" on lift_logs for all using (true) with check (true);
